统一 Session 数据模型

Yeek 的中间格式 — 连接不同 agent 的通用语言

SQLite 表结构

-- 会话元数据
CREATE TABLE sessions (
  id              TEXT PRIMARY KEY,  -- yeek 统一 ID (UUIDv7)
  source_agent    TEXT NOT NULL,      -- "claude_code" | "codex" | ...
  source_id      TEXT NOT NULL,      -- 原始 agent 的 session ID
  title           TEXT,            -- 首条用户消息 / 自动摘要
  project_path    TEXT,          -- 工作目录
  git_branch      TEXT,
  model           TEXT,            -- 使用的模型名
  status          TEXT DEFAULT 'active', -- active | completed | archived
  import_offset   INTEGER DEFAULT 0, -- 已解析到的 JSONL 行号
  source_mtime    REAL,            -- 原始文件 mtime，用于增量判断
  tags            TEXT,            -- JSON array of tags
  token_count     INTEGER DEFAULT 0,
  msg_count       INTEGER DEFAULT 0,
  created_at      TEXT NOT NULL,
  updated_at      TEXT NOT NULL
);

-- 消息内容（压缩存储）
CREATE TABLE messages (
  id              INTEGER PRIMARY KEY AUTOINCREMENT,
  session_id      TEXT NOT NULL REFERENCES sessions(id),
  seq             INTEGER NOT NULL,     -- 消息在会话中的顺序
  role            TEXT NOT NULL,      -- user | assistant | system | tool
  content         BLOB,            -- zstd 压缩的 JSON 内容
  content_type    TEXT DEFAULT 'text', -- text | tool_call | tool_result | thinking
  parent_id       INTEGER,          -- 支持分支（Claude Code 的 parentUuid）
  source_type     TEXT,            -- 原始事件类型（agent-specific）
  source_raw      BLOB,            -- zstd 压缩的原始 JSON（保留 agent 特有字段）
  timestamp        TEXT NOT NULL,
  UNIQUE(session_id, seq)
);

-- 全文搜索索引
CREATE VIRTUAL TABLE messages_fts USING fts5(
  content_text,  -- 解压后的纯文本用于搜索
  session_id UNINDEXED,
  content='messages',
  content_rowid='id'
  -- 使用 triggers 同步
                  -- 或在导入时批量写入
);

-- Agent 适配器注册
CREATE TABLE adapters (
  name           TEXT PRIMARY KEY,
  version         TEXT NOT NULL,
  last_sync_at    TEXT,
  watch_paths    TEXT        -- JSON array of watched directories
);

关键设计决策

双内容存储

content 是统一格式的消息体，source_raw 保留原始 agent 数据。转换时不丢信息，回写时可以还原 agent 特有字段。

增量追踪

import_offset + source_mtime 支持增量导入。文件没变就跳过，变了只读新增行。

FTS5 外部内容

使用外部内容模式，搜索文本不重复存储，只建倒排索引。节省空间且查询高效。
